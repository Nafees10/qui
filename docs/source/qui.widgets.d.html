12 /// To contain another widget in "margins"
13 /// 
14 /// set margins using the `marginLeft/Right/Top/Bottom` properties, or simply using `margin` - default margins are zero
15 /// set the character to write in margin using `marginCharTop/Bottom/Left/Right` or just `marginChar` - default is space character
16 class ContainerWidget : QWidget{
17 private:
18 	QWidget childWidget; // the widget to be "contained"
19 	bool childWidgetIsActive;
20 	uinteger mTop, mBottom, mLeft, mRight; // margins
21 	char mCharTop, mCharBottom, mCharLeft, mCharRight;
22 
23 	/// called by `update` to draw margins
24 	void drawMargins(Matrix display){
25 		char[] emptyLine;
26 		// top
27 		emptyLine.length = widgetSize.width;
28 		emptyLine[] = mCharTop;
29 		display.moveTo(0, 0);
30 		for (uinteger i = 0; i < mTop; i ++){
31 			display.write(emptyLine, textColor, backgroundColor);
32 		}
33 		// bottom
34 		emptyLine[] = mCharBottom;
35 		display.moveTo(0, widgetSize.height - mBottom);
36 		for (uinteger i = 0; i < mBottom; i ++){
37 			display.write(emptyLine, textColor, backgroundColor);
38 		}
39 		//left
40 		emptyLine.length = mLeft;
41 		emptyLine[] = mCharLeft;
42 		for (uinteger i = 0, count = widgetSize.height-(mTop+mBottom); i < count; i ++){
43 			display.moveTo(0, mTop+i);
44 			display.write(emptyLine, textColor, backgroundColor);
45 		}
46 		// right
47 		emptyLine.length = mRight;
48 		emptyLine[] = mCharRight;
49 		for (uinteger i = 0, count = widgetSize.height-(mTop+mBottom); i < count; i ++){
50 			display.moveTo(widgetSize.width-(mRight), mTop+i);
51 			display.write(emptyLine, textColor, backgroundColor);
52 		}
53 	}
54 	/// called by margin-changing-properties to calculate new minHeight & minWidth to fit the widget in
55 	void calculateMinSize(){
56 		// first minHeight
57 		uinteger minHeight = (childWidget !is null ? childWidget.size.minHeight : 0) + mTop + mBottom;
58 		if (widgetSize.minHeight < minHeight){
59 			widgetSize.minHeight = minHeight;
60 		}
61 		// then width
62 		uinteger minWidth = (childWidget !is null ? childWidget.size.minWidth : 0) + mLeft + mRight;
63 		if (widgetSize.minWidth < minWidth){
64 			widgetSize.minWidth = minWidth;
65 		}
66 	}
67 public:
68 	/// background and text colors
69 	RGB backgroundColor, textColor;
70 	this (){
71 		textColor = DEFAULT_TEXT_COLOR;
72 		backgroundColor = DEFAULT_BACK_COLOR;
73 		marginChar = ' ';
74 		margin = 0;
75 		// this widget doesnt want input, but it's child-widget may want, but child widget is registered separately, so 
76 		// this widget does not want input
77 		widgetWantsInput = false;
78 	}
79 
80 	override bool update(Matrix display){
81 		if (needsUpdate || (childWidget !is null && childWidget.visible && childWidget.needsUpdate)){
82 			// then the widget
83 			Matrix wDisplay = new Matrix(childWidget.size.width, childWidget.size.height);
84 			if (childWidget.update(wDisplay)){
85 				display.insert(wDisplay, mLeft, mTop);
86 			}
87 			.destroy(wDisplay);
88 			// draw the margins, only if "this" widget needs update, meaning: only if it resized or something messed up margins
89 			if (needsUpdate){
90 				drawMargins(display);
91 				needsUpdate = false;
92 			}
93 
94 			return true;
95 		}else{
96 			return false;
97 		}
98 	}
99 
100 	override public void resizeEvent(){
101 		super.resizeEvent();
102 		// resize and reposition childWidget if any
103 		if (childWidget !is null){
104 			childWidget.size.width = widgetSize.width - (mLeft + mRight);
105 			childWidget.size.height = widgetSize.height - (mTop + mBottom);
106 			childWidget.resizeEvent();
107 
108 			childWidget.position.x = this.position.x + mLeft;
109 			childWidget.position.y = this.position.y + mTop;
110 		}
111 	}

257 	RGB backgroundColor, barColor;

270 			r = true;
271 			uinteger filled = ratioToRaw(done, max, widgetSize.width);
272 			char[] bar;
273 			bar.length = widgetSize.width;
274 			bar[0 .. bar.length] = ' ';
275 			for (uinteger i = 0; i < widgetSize.height; i++){
276 				display.write(bar[0 .. filled], barColor, barColor);
277 				display.write(bar[filled .. bar.length], barColor, backgroundColor);
278 			}
279 			needsUpdate = false;
280 		}
281 		return r;
282 	}

309 /// To get single-line input from keyboard
310 class EditLineWidget : QWidget{
311 private:
312 	char[] inputText;
313 	uinteger cursorX;
314 	uinteger scrollX = 0;//amount of chars that won't be displayed because of not enough space
315 	
316 	void reScroll(){
317 		//check if is within length of line
318 		if (cursorX > inputText.length){
319 			cursorX = inputText.length;
320 		}
321 		uinteger w = widgetSize.width - widgetCaption.length;
322 		//now calculate scrollX, if it needs to be increased
323 		if ((scrollX + w < cursorX || scrollX + w >= cursorX)){
324 			if (cursorX <= w){
325 				scrollX = 0;
326 			}else{
327 				scrollX = cursorX - (w/2);
328 			}
329 		}
330 	}
331 	/// used by widget itself to set cursor
332 	void setCursor(){
333 		termInterface.setCursorPos(
334 			Position(
335 				(cursorX - scrollX)+widgetCaption.length, // x
336 				0), // y
337 			this);
338 	}
339 	///shortens caption if too long
340 	void shortenCaption(){
341 		if (widgetSize.width - widgetCaption.length < 4){
342 			widgetCaption.length = widgetSize.width - 4;
343 		}
344 	}
345 public:
346 	/// background, text, caption, and caption's background colors
347 	RGB backgroundColor, textColor, captionTextColor, captionBackgroundColor;
348 	this(string wCaption = "", string inputTxt = ""){
349 		inputText = cast(char[])inputTxt;
350 		widgetCaption = wCaption;
351 		shortenCaption;
352 		//specify min/max
353 		widgetSize.minWidth = 1;
354 		widgetSize.minHeight = 1;
355 		widgetSize.maxHeight = 1;
356 		// this widget wants Tab key
357 		widgetWantsTab = true;
358 		// and input too, obvious
359 		widgetWantsInput = true;
360 		// and needs to show the cursor too
361 		widgetShowCursor = true;
362 
363 		textColor = DEFAULT_TEXT_COLOR;
364 		backgroundColor = DEFAULT_BACK_COLOR;
365 		captionTextColor = DEFAULT_TEXT_COLOR;
366 		captionBackgroundColor = DEFAULT_BACK_COLOR;
367 	}
368 
369 	override bool update(Matrix display){
370 		bool r = false;
371 		if (needsUpdate){
372 			r = true;
373 			//make sure there's enough space
374 			if (widgetSize.width > widgetCaption.length){
375 				//draw the caption
376 				display.write(cast(char[])widgetCaption, captionTextColor, captionBackgroundColor);
377 				//draw the inputText
378 				uinteger width = widgetSize.width - widgetCaption.length;
379 				//fit the line into screen, i.e check if only a part of it will be displayed
380 				if (inputText.length >= width+scrollX){
381 					//display only partial line
382 					display.write(inputText[scrollX .. scrollX + width], textColor, backgroundColor);
383 				}else{
384 					char[] emptyLine;
385 					emptyLine.length = width;
386 					emptyLine[] = ' ';
387 					//either the line is small enough to fit, or 0-length
388 					if (inputText.length <= scrollX || inputText.length == 0){
389 						//just write the bgColor
390 						display.write(emptyLine, textColor, backgroundColor);
391 					}else{
392 						display.write(inputText[scrollX .. inputText.length], textColor, backgroundColor);
393 						//write the bgColor
394 						display.write(emptyLine[inputText.length - scrollX .. emptyLine.length], textColor, backgroundColor);
395 					}
396 				}
397 			}else{
398 				// be sad, there's not enough space to draw
399 				if (this.widgetSize.width >= 2){
400 					display.write(cast(char[])":(", textColor, backgroundColor);
401 				}
402 			}
403 			needsUpdate = false;
404 		}
405 		setCursor();
406 		return r;
407 	}
408 	
409 	override void mouseEvent(MouseClick mouse){
410 		super.mouseEvent(mouse);
411 		if (mouse.mouseButton == mouse.Button.Left){
412 			needsUpdate = true;
413 			//move cursor to that pos
414 			if (mouse.x > widgetCaption.length && mouse.x < widgetCaption.length + inputText.length){
415 				cursorX = mouse.x - (widgetCaption.length + scrollX);
416 			}
417 		}
418 		reScroll;
419 	}
420 	override void keyboardEvent(KeyPress key){
421 		super.keyboardEvent(key);
422 		if (key.isChar){
423 			needsUpdate = true;
424 			//insert that key
425 			if (key.key != '\b' && key.key != '\n'){
426 				if (cursorX == inputText.length){
427 					//insert at end
428 					inputText ~= cast(char)key.key;
429 				}else{
430 					inputText = inputText.insertElement([cast(char)key.key], cursorX);
431 				}
432 				cursorX ++;
433 			}else if (key.key == '\b'){
434 				//backspace
435 				if (cursorX > 0){
436 					if (cursorX == inputText.length){
437 						inputText.length --;
438 					}else{
439 						inputText = inputText.deleteElement(cursorX-1);
440 					}
441 					cursorX --;
442 				}
443 			}
444 		}else{
445 			if (key.key == key.NonCharKey.LeftArrow && cursorX > 0){
446 				needsUpdate = true;
447 				cursorX --;
448 			}else if (key.key == key.NonCharKey.RightArrow && cursorX < inputText.length){
449 				needsUpdate = true;
450 				cursorX ++;
451 			}else if (key.key == key.NonCharKey.Delete && cursorX < inputText.length){
452 				inputText = inputText.deleteElement(cursorX);
453 			}
454 		}
455 		reScroll;
456 	}
457 	///The text that has been input-ed.
458 	@property string text(){
459 		return cast(string)inputText;
460 	}
461 	///The text that has been input-ed.
462 	@property string text(string newText){
463 		inputText = cast(char[])newText;
464 		// force an update
465 		termInterface.forceUpdate();
466 		return cast(string)inputText;
467 	}
468 	/// caption of the widget. setter
469 	override @property string caption(string newCaption){
470 		needsUpdate = true;
471 		widgetCaption = newCaption;
472 		shortenCaption;
473 		// force an update
474 		termInterface.forceUpdate();
475 		return widgetCaption;
476 	}
477 	/// override resize to shorten caption
478 	override void resizeEvent(){
479 		super.resizeEvent;
480 		shortenCaption;
481 	}
482 }

484 /// Can be used as a simple text editor, or to just display text
485 class MemoWidget : QWidget{
486 private:
487 	List!string widgetLines;
488 	uinteger scrollX, scrollY;
489 	uinteger cursorX, cursorY;
490 	bool writeProtected = false;
491 	// used by widget itseld to recalculate scrolling
492 	void reScroll(){
493 		//calculate scrollY first
494 		scrollY = 0;
495 		// scrollY + widgetSize.height  -->  last visible line+1
496 		if (cursorY >= scrollY + widgetSize.height || cursorY < scrollY){
497 			scrollY = (cursorY - widgetSize.height) + 1;
498 		}
499 		//now time for scrollX
500 		//check if is within length of line
501 		uinteger len = readLine(cursorY).length;
502 		if (cursorX > len){
503 			cursorX = len;
504 		}
505 		//now calculate scrollX, if it needs to be increased
506 		if (scrollX + widgetSize.width < cursorX || scrollX + widgetSize.width >= cursorX){
507 			if (cursorX < widgetSize.width){
508 				scrollX = 0;
509 			}else{
510 				scrollX = cursorX - widgetSize.width;
511 			}
512 		}
513 	}
514 	// used by widget itself to set cursor
515 	void setCursor(){
516 		termInterface.setCursorPos(
517 			Position(
518 				cursorX - scrollX, // x
519 				cursorY - scrollY //  y
520 				),
521 			this);
522 	}
523 	// used by widget itself to move cursor
524 	void moveCursor(uinteger x, uinteger y){
525 		cursorX = x;
526 		cursorY = y;
527 		
528 		if (cursorY > lineCount){
529 			cursorY = lineCount-1;
530 		}
531 		if (cursorX > readLine(cursorY).length){
532 			cursorX = readLine(cursorY).length;
533 		}
534 	}
535 	/// Reads a line from widgetLines
536 	string readLine(uinteger index){
537 		if (index == widgetLines.length){
538 			return "";
539 		}
540 		return widgetLines.read(index);
541 	}
542 	/// overwrites a line
543 	void overwriteLine(uinteger index, string line){
544 		if (index == widgetLines.length){
545 			widgetLines.add(line);
546 		}else{
547 			widgetLines.set(index,line);
548 		}
549 	}
550 	/// deletes a line
551 	void removeLine(uinteger index){
552 		if (index < widgetLines.length){
553 			widgetLines.remove(index);
554 		}
555 	}
556 	/// inserts a line
557 	void insertLine(uinteger index, string line){
558 		if (index == widgetLines.length){
559 			widgetLines.add(line);
560 		}else{
561 			widgetLines.insert(index, line);
562 		}
563 	}
564 	/// adds a line
565 	void addLine(string line){
566 		widgetLines.add(line);
567 	}
568 	/// returns lines count
569 	@property uinteger lineCount(){
570 		return widgetLines.length+1;
571 	}
572 public:
573 	// background and text colors
574 	RGB backgroundColor, textColor;
575 	this(bool readOnly = false){
576 		widgetLines = new List!string;
577 		scrollX = 0;
578 		scrollY = 0;
579 		cursorX = 0;
580 		cursorY = 0;
581 		writeProtected = readOnly;//cause if readOnly, then writeProtected = true also
582 
583 		// this widget wants Tab key
584 		widgetWantsTab = true;
585 		// and input too, obviously
586 		widgetWantsInput = true;
587 		// and the cursor should be visible
588 		widgetShowCursor = true;
589 
590 		textColor = DEFAULT_TEXT_COLOR;
591 		backgroundColor = DEFAULT_BACK_COLOR;
592 	}
593 	~this(){
594 		.destroy(widgetLines);
595 	}
596 	
597 	override bool update(Matrix display){
598 		bool r = false;
599 		if (needsUpdate){
600 			r = true;
601 			//check if there's lines to be displayed
602 			uinteger count = lineCount;
603 			if (count > 0){
604 				//write lines to memo
605 				char[] line;
606 				for (uinteger i = scrollY; i < count; i++){
607 					//echo current line
608 					line = cast(char[])readLine(i);
609 					//fit the line into screen, i.e check if only a part of it will be displayed
610 					if (line.length >= widgetSize.width+scrollX){
611 						//display only partial line
612 						display.write(line[scrollX .. scrollX + widgetSize.width], textColor, backgroundColor);
613 					}else{
614 						//either the line is small enough to fit, or 0-length
615 						if (line.length <= scrollX || line.length == 0){
616 							//just write the bgColor
617 							display.fillLine(' ', textColor, backgroundColor);
618 						}else{
619 							display.write(line[scrollX .. line.length], textColor, backgroundColor);
620 							//write the bgColor
621 							display.fillLine(' ',textColor,backgroundColor);
622 						}
623 					}
624 					//check if is at end
625 					if (i-scrollY >= widgetSize.height){
626 						break;
627 					}
628 				}
629 				//fill empty space with emptyLine
630 				display.fillMatrix(' ',textColor,backgroundColor);
631 			}
632 			needsUpdate = false;
633 		}
634 		setCursor();
635 		return r;
636 	}
637 	
638 	override void mouseEvent(MouseClick mouse){
639 		super.mouseEvent(mouse);
640 		//calculate mouse position, relative to scroll
641 		mouse.x = mouse.x + scrollX;
642 		mouse.y = mouse.y + scrollY;
643 		if (mouse.mouseButton == mouse.Button.Left){
644 			needsUpdate = true;
645 			moveCursor(mouse.x, mouse.y);
646 		}else if (mouse.mouseButton == mouse.Button.ScrollDown){
647 			if (cursorY+1 < lineCount){
648 				needsUpdate = true;
649 				moveCursor(cursorX, cursorY + 4);
650 				reScroll();
651 			}
652 		}else if (mouse.mouseButton == mouse.Button.ScrollUp){
653 			if (cursorY > 0){
654 				needsUpdate = true;
655 				if (cursorY < 4){
656 					moveCursor(cursorX, 0);
657 				}else{
658 					moveCursor(cursorX, cursorY - 4);
659 				}
660 				reScroll();
661 			}
662 		}
663 	}
664 	
665 	override void keyboardEvent(KeyPress key){
666 		super.keyboardEvent(key);
667 		if (key.isChar){
668 			if (!writeProtected){
669 				needsUpdate = true;
670 				string currentLine = readLine(cursorY);
671 				//check if backspace
672 				if (key.key == '\b'){
673 					//make sure that it's not the first line, first line cannot be removed
674 					if (cursorY > 0){
675 						//check if has to remove a '\n'
676 						if (cursorX == 0){
677 							cursorY --;
678 							//if line's not empty, append it to previous line
679 							cursorX = readLine(cursorY).length;
680 							if (currentLine != ""){
681 								//else, append this line to previous
682 								overwriteLine(cursorY, readLine(cursorY)~currentLine);
683 							}
684 							removeLine(cursorY+1);
685 						}else{
686 							overwriteLine(cursorY, cast(string)deleteElement(cast(char[])currentLine,cursorX-1));
687 							cursorX --;
688 						}
689 					}else if (cursorX > 0){
690 						overwriteLine(cursorY, cast(string)deleteElement(cast(char[])currentLine,cursorX-1));
691 						cursorX --;
692 					}
693 					
694 				}else if (key.key == '\n'){
695 					//insert a newline
696 					//if is at end, just add it
697 					bool atEnd = false;
698 					if (cursorY >= lineCount - 1){
699 						atEnd = true;
700 					}
701 					if (cursorX == readLine(cursorY).length){
702 						if (atEnd){
703 							widgetLines.add("");
704 						}else{
705 							insertLine(cursorY + 1,"");
706 						}
707 					}else{
708 						string[2] line;
709 						line[0] = readLine(cursorY);
710 						line[1] = line[0][cursorX .. line[0].length];
711 						line[0] = line[0][0 .. cursorX];
712 						overwriteLine(cursorY, line[0]);
713 						if (atEnd){
714 							widgetLines.add(line[1]);
715 						}else{
716 							insertLine(cursorY + 1, line[1]);
717 						}
718 					}
719 					cursorY ++;
720 					cursorX = 0;
721 				}else if (key.key == '\t'){
722 					//convert it to 4 spaces
723 					overwriteLine(cursorY, cast(string)insertElement(cast(char[])currentLine,cast(char[])"    ",cursorX));
724 					cursorX += 4;
725 				}else{
726 					//insert that char
727 					overwriteLine(cursorY, cast(string)insertElement(cast(char[])currentLine,[cast(char)key.key],cursorX));
728 					cursorX ++;
729 				}
730 			}
731 		}else{
732 			if (key.key == key.NonCharKey.Delete){
733 				needsUpdate = true;
734 				//check if is deleting \n
735 				if (cursorX == readLine(cursorY).length && cursorY+1 < lineCount){
736 					//merge next line with this one
737 					char[] line = cast(char[])readLine(cursorY)~readLine(cursorY+1);
738 					overwriteLine(cursorY, cast(string)line);
739 					//remove next line
740 					removeLine(cursorY+1);
741 				}else if (cursorX < readLine(cursorY).length){
742 					char[] line = cast(char[])readLine(cursorY);
743 					line = line.deleteElement(cursorX);
744 					overwriteLine(cursorY, cast(string)line);
745 				}
746 			}else if (key.key == key.NonCharKey.DownArrow){
747 				if (cursorY+1 < lineCount){
748 					needsUpdate = true;
749 					cursorY ++;
750 				}
751 			}else if (key.key == key.NonCharKey.UpArrow){
752 				if (cursorY > 0){
753 					needsUpdate = true;
754 					cursorY --;
755 				}
756 			}else if (key.key == key.NonCharKey.LeftArrow){
757 				if ((cursorY >= 0 && cursorX > 0) || (cursorY > 0 && cursorX == 0)){
758 					needsUpdate = true;
759 					if (cursorX == 0){
760 						cursorY --;
761 						cursorX = readLine(cursorY).length;
762 					}else{
763 						cursorX --;
764 					}
765 				}
766 			}else if (key.key == key.NonCharKey.RightArrow){
767 				needsUpdate = true;
768 				if (cursorX == readLine(cursorY).length){
769 					if (cursorY+1 < lineCount){
770 						cursorX = 0;
771 						cursorY ++;
772 						scrollX = 0;
773 					}
774 				}else{
775 					cursorX ++;
776 				}
777 			}
778 		}
779 		// I'll use this this time not to move the cursor, but to fix the cursor position
780 		moveCursor(cursorX,cursorY);
781 		reScroll();
782 	}

784 	///returns a list of lines in memo
785 	///
786 	///To modify the content, just modify it in the returned list
787 	///
788 	///class `List` is defined in `utils.lists.d`
789 	@property List!string lines(){
790 		return widgetLines;
791 	}
792 	///Returns true if memo's contents cannot be modified, by user
793 	@property bool readOnly(){
794 		return writeProtected;
795 	}
796 	///sets whether to allow modifying of contents (false) or not (true)
797 	@property bool readOnly(bool newPermission){
798 		writeProtected = newPermission;
799 		// force an update

815 	uinteger stringLineCount(string s){
816 		uinteger width = widgetSize.width;
817 		uinteger i, widthTaken = 0, count = 1;
818 		for (i = 0; i < s.length; i++){
819 			widthTaken ++;
820 			if (s[i] == '\n' || widthTaken >= width){
821 				count ++;
822 				widthTaken = 0;
823 			}
824 			if (widthTaken >= width){
825 				count ++;
826 			}
827 		}
828 		return count;
829 	}
830 	string stripString(string s, uinteger height){
831 		char[] r;
832 		uinteger width = widgetSize.width;
833 		uinteger i, widthTaken = 0, count = 1;
834 		for (i = 0; i < s.length; i++){
835 			widthTaken ++;
836 			if (s[i] == '\n' || widthTaken >= width){
837 				count ++;
838 				widthTaken = 0;
839 			}
840 			if (count > height){
841 				r.length = i;
842 				r[0 .. i] = s[0 .. i];
843 				break;
844 			}
845 		}
846 		return cast(string)r;
847 	}

850 	RGB backgroundColor, textColor;
851 	this(uinteger maxLen=100){
852 		max = maxLen;
853 		logs = new LinkedList!string;
854 
855 		textColor = DEFAULT_TEXT_COLOR;
856 		backgroundColor = DEFAULT_BACK_COLOR;
857 	}
858 	~this(){
859 		logs.destroy;
860 	}

862 	override public bool update(Matrix display){
863 		bool r = false;
864 		if (needsUpdate){
865 			r = true;
866 			//get list of messages
867 			string[] messages = logs.toArray;
868 			//messages = messages.arrayReverse;
869 			//set colors
870 			display.setColors(textColor, backgroundColor);
871 			//determine how many of them will be displayed
872 			uinteger count;//right now, it's used to store number of lines used
873 			uinteger i;
874 			if (messages.length>0){
875 				for (i=messages.length-1; i>=0; i--){
876 					count += stringLineCount(messages[i]);
877 					if (count > widgetSize.height){
878 						messages = messages[i+1 .. messages.length];
879 						//try to insert part of the last message
880 						uinteger thisCount = stringLineCount(messages[i]);
881 						count -= thisCount;
882 						if (count < widgetSize.height){
883 							messages ~= [stripString(messages[i], widgetSize.height - count)];
884 						}
885 						break;
886 					}
887 					if (i==0){
888 						break;
889 					}
890 				}
891 			}
892 			//write them
893 			for (i = 0; i < messages.length; i++){
894 				display.write(cast(char[])messages[i], textColor, backgroundColor);
895 				//add newline
896 				display.moveTo(0, display.writePos.y+1);
897 			}
898 			needsUpdate = false;
899 		}
900 		return r;
901 	}
