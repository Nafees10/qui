302 ///Used to place widgets in an order (i.e vertical or horizontal)
303 ///
304 ///The QTerminal is also a layout, basically.
305 ///
306 ///Name in theme: 'layout';
307 class QLayout : QWidget{
308 private:
309 	// array of all the widgets that have been added to this layout
310 	QWidget[] widgetList;
311 	// contains reference to the active widget, null if no active widget
312 	QWidget activeWidget;
313 	// stores the layout type, horizontal or vertical
314 	LayoutDisplayType layoutType;
315 	
316 	// background color
317 	RGBColor bgColor;
318 	// foreground color
319 	RGBColor textColor;
320 	// stores whether an update is in progress
321 	bool isUpdating = false;
322 	
323 	// recalculates the size of every widget inside layout
324 	void recalculateWidgetsSize(LayoutDisplayType T)(QWidget[] widgets, uinteger totalSpace, uinteger totalRatio){
325 		static if (T != LayoutDisplayType.Horizontal && T != LayoutDisplayType.Vertical){
326 			assert(false);
327 		}
328 		uinteger repeat;
329 		do{
330 			repeat = false;
331 			foreach(i, widget; widgets){
332 				if (widget.visible){
333 					// calculate width or height
334 					uinteger newSpace = ratioToRaw(widget.sizeRatio, totalRatio, totalSpace);
335 					uinteger calculatedSpace = newSpace;
336 					//apply size
337 					static if (T == LayoutDisplayType.Horizontal){
338 						widget.size.height = widgetSize.height;
339 						widget.size.width = newSpace;
340 						newSpace = widget.size.width;
341 					}else{
342 						widget.size.width = widgetSize.width;
343 						widget.size.height = newSpace;
344 						newSpace = widget.size.height;
345 					}
346 					// let the know it was resized, useful if the widget is a layout. This marks `needsUpdate` true in the widget
347 					widget.resize;
348 					if (newSpace != calculatedSpace){
349 						totalRatio -= widget.sizeRatio;
350 						totalSpace -= newSpace;
351 						widgets = widgets.dup.deleteElement(i);
352 						repeat = true;
353 						break;
354 					}
355 					// check if there's enough space to contain that widget
356 					if (newSpace > totalSpace){
357 						newSpace = 0;
358 						widget.visible = false;
359 					}
360 				}
361 			}
362 		} while (repeat);
363 	}
364 	/// calculates and assigns widgets positions based on their sizes
365 	void recalculateWidgetsPosition(LayoutDisplayType T)(QWidget[] widgets){
366 		static if (T != LayoutDisplayType.Horizontal && T != LayoutDisplayType.Vertical){
367 			assert(false);
368 		}
369 		uinteger previousSpace = (T == LayoutDisplayType.Horizontal ? widgetPosition.x : widgetPosition.y);
370 		uinteger fixedPoint = (T == LayoutDisplayType.Horizontal ? widgetPosition.y : widgetPosition.x);
371 		foreach(widget; widgets){
372 			if (widget.visible){
373 				static if (T == LayoutDisplayType.Horizontal){
374 					widget.position.y = widgetPosition.y;
375 					widget.position.x = previousSpace;
376 					previousSpace += widget.size.width;
377 				}else{
378 					widget.position.x = widgetPosition.x;
379 					widget.position.y = previousSpace;
380 					previousSpace += widget.size.height;
381 				}
382 			}
383 		}
384 	}
385 	
386 public:
387 	this(LayoutDisplayType type){
388 		layoutType = type;
389 		activeWidget = null;
390 	}

392 	/// adds (appends) a widget to the widgetList, and makes space for it
393 	/// 
394 	/// If there a widget is too large, it's marked as not visible
395 	void addWidget(QWidget widget){
396 		widget.onForceUpdate = forceUpdate;
397 		//add it to array
398 		widgetList.length++;
399 		widgetList[widgetList.length-1] = widget;
400 		//recalculate all widget's size to adjust
401 		resize();
402 	}
403 	/// adds (appends) widgets to the widgetList, and makes space for them
404 	/// 
405 	/// If there a widget is too large, it's marked as not visible
406 	void addWidget(QWidget[] widgets){
407 		foreach(widget; widgets){
408 			widget.onForceUpdate = forceUpdate;
409 		}
410 		// add to array
411 		widgetList ~= widgets.dup;
412 		//resize
413 		resize();
414 	}
415 	
416 	/// Recalculates size and position for all visible widgets
417 	/// If a widget is too large to fit in, it's visibility is marked false
418 	override void resize(){
419 		//disable update during size change, saves CPU and time
420 		isUpdating = true;
421 		uinteger ratioTotal;
422 		foreach(w; widgetList){
423 			if (w.visible()){
424 				ratioTotal += w.sizeRatio;
425 			}
426 		}
427 		if (layoutType == LayoutDisplayType.Horizontal){
428 			recalculateWidgetsSize!(LayoutDisplayType.Horizontal)(widgetList, widgetSize.width, ratioTotal);
429 			recalculateWidgetsPosition!(LayoutDisplayType.Horizontal)(widgetList);
430 		}else{
431 			recalculateWidgetsSize!(LayoutDisplayType.Vertical)(widgetList, widgetSize.height, ratioTotal);
432 			recalculateWidgetsPosition!(LayoutDisplayType.Vertical)(widgetList);
433 		}
434 		isUpdating = false;
435 	}
436 	
437 	override void mouseEvent(MouseClick mouse){
438 		super.mouseEvent(mouse);
439 		//remove access to cursor from previous active widget
440 		if (activeWidget){
441 			activeWidget.onCursorPosition = null;
442 		}
443 		foreach (widget; widgetList){
444 			if (widget.visible){
445 				Position p = widget.position;
446 				Size s = widget.size;
447 				//check x-axis
448 				if (mouse.x >= p.x && mouse.x < p.x + s.width){
449 					//check y-axis
450 					if (mouse.y >= p.y && mouse.y < p.y + s.height){
451 						//give access to cursor position
452 						widget.onCursorPosition = cursorPos;
453 						//call mouseEvent
454 						widget.mouseEvent(mouse);
455 						//mark this widget as active
456 						activeWidget = widget;
457 						break;
458 					}
459 				}
460 			}
461 		}
462 	}
463 	override void keyboardEvent(KeyPress key){
464 		super.keyboardEvent(key);
465 		//check active widget, call keyboardEvent
466 		if (activeWidget){
467 			activeWidget.keyboardEvent(key);
468 		}
469 	}
470 	override bool update(ref Matrix display){
471 		bool updated = false;
472 		//check if already updating, case yes, return false
473 		if (!isUpdating){
474 			isUpdating = true;
475 			//go through all widgets, check if they need update, update them
476 			Matrix wDisplay = new Matrix(1,1);
477 			foreach(widget; widgetList){
478 				if (widget.visible){
479 					wDisplay.changeSize(widget.size.width, widget.size.height);
480 					wDisplay.resetWritePosition();
481 					if (widget.update(wDisplay)){
482 						display.insert(wDisplay, widget.position.x, widget.position.y);
483 						updated = true;
484 					}
485 					wDisplay.clear();
486 				}
487 			}
488 			isUpdating = false;
489 		}else{
490 			return false;
491 		}
492 		return updated;
493 	}
494 
495 	// override setOnForceUpdate to change it for all child widgets as well
496 	override @property bool delegate() onForceUpdate(bool delegate() newOnForceUpdate){
497 		// change it for all child widgets
498 		foreach(widget; widgetList){
499 			widget.onForceUpdate = newOnForceUpdate;
500 		}
501 		// change it for itself
502 		return super.onForceUpdate(newOnForceUpdate);
503 	}
504 }

507 /// 
508 /// All widgets, receives events, runs UI loop...

512 private:
513 	Terminal terminal;
514 	RealTimeConsoleInput input;
515 	Matrix termDisplay;
516 	
517 	Position cursorPos;
518 	
519 	bool isRunning = false;
520 public:
521 	this(string caption = "QUI Text User Interface", LayoutDisplayType displayType = LayoutDisplayType.Vertical){
522 		super(displayType);
523 		//create terminal & input
524 		terminal = Terminal(ConsoleOutputType.cellular);
525 		input = RealTimeConsoleInput(&terminal, ConsoleInputFlags.allInputEvents);
526 		terminal.showCursor();
527 		//init vars
528 		widgetSize.height = terminal.height;
529 		widgetSize.width = terminal.width;
530 		widgetCaption = caption;
531 		//set caption
532 		terminal.setTitle(widgetCaption);
533 		//create display matrix
534 		termDisplay = new Matrix(widgetSize.width, widgetSize.height);
535 		termDisplay.setColors(textColor, bgColor);
536 	}

545 	}

554 		super.mouseEvent(mouse);

556 		if (activeWidget){
557 			activeWidget.onCursorPosition = null;
558 		}
559 		foreach (widget; widgetList){
560 			if (widget.visible){
561 				Position p = widget.position;
562 				Size s = widget.size;
563 				//check x-axis
564 				if (mouse.x >= p.x && mouse.x < p.x + s.width){
565 					//check y-axis
566 					if (mouse.y >= p.y && mouse.y < p.y + s.height){
567 						//give access to cursor position
568 						widget.onCursorPosition = &setCursorPos;
569 						//call mouseEvent
570 						widget.mouseEvent(mouse);

574 					}

580 	/// Use this instead of `update` to forcefully update the terminal
581 	/// 
582 	/// returns true if at least one widget was updated, false if nothing was updated
583 	bool updateDisplay(){
584 		if (isRunning && !isUpdating){
585 			bool r = update(termDisplay);
586 			if (r){
587 				terminal.moveTo(0, 0);
588 				termDisplay.flushToTerminal(this);
589 			}
590 			//set cursor position
591 			terminal.moveTo(cast(int)cursorPos.x, cast(int)cursorPos.y);
592 			terminal.showCursor();
593 			return r;
594 		}else{
595 			return false;
596 		}
597 	}
598 	
599 	/// starts the UI loop
600 	void run(){
601 		InputEvent event;
602 		isRunning = true;
603 		//resize all widgets
604 		resize();
605 		//draw the whole thing
606 		updateDisplay();
607 		while (isRunning){
608 			event = input.nextEvent;
609 			//check event type
610 			if (event.type == event.Type.KeyboardEvent){
611 				// prevent any updates during event, the update will be done at end
612 				isUpdating = true;
613 				KeyPress kPress;
614 				kPress.key = event.get!(event.Type.KeyboardEvent).which;
615 				this.keyboardEvent(kPress);
616 				// now it can update
617 				isUpdating = false;
618 				updateDisplay;
619 			}else if (event.type == event.Type.MouseEvent){
620 				// prevent any updates during event, the update will be done at end
621 				isUpdating = true;
622 				MouseEvent mEvent = event.get!(event.Type.MouseEvent);
623 				MouseClick mPos;
624 				mPos.x = mEvent.x;
625 				mPos.y = mEvent.y;
626 				switch (mEvent.buttons){
627 					case MouseEvent.Button.Left:
628 						mPos.mouseButton = mPos.Button.Left;
629 						break;
630 					case MouseEvent.Button.Right:
631 						mPos.mouseButton = mPos.Button.Right;
632 						break;
633 					case MouseEvent.Button.ScrollUp:
634 						mPos.mouseButton = mPos.Button.ScrollUp;
635 						break;
636 					case MouseEvent.Button.ScrollDown:
637 						mPos.mouseButton = mPos.Button.ScrollDown;
638 						break;
639 					default:
640 						continue;
641 				}
642 				this.mouseEvent(mPos);
643 				// now it can update
644 				isUpdating = false;
645 				updateDisplay;
646 			}else if (event.type == event.Type.SizeChangedEvent){
647 				//change matrix size
648 				termDisplay.changeSize(cast(uinteger)terminal.width, cast(uinteger)terminal.height);
649 				termDisplay.setColors(textColor, bgColor);
650 				//update self size
651 				terminal.updateSize;
652 				widgetSize.height = terminal.height;
653 				widgetSize.width = terminal.width;
654 				this.clear;
655 				//call size change on all widgets
656 				resize();
657 				updateDisplay;
658 			}else if (event.type == event.Type.UserInterruptionEvent){
659 				//die here
660 				terminal.clear;
661 				isRunning = false;
662 				break;
663 			}
664 		}
665 		//in case an exception prevents it from being set to false before
666 		isRunning = false;
667 	}

712 /// Used to manage the characters on the terminal
713 class Matrix{
714 private:
715 	struct Display{
716 		uinteger x, y;
717 		RGBColor bgColor, textColor;
718 		char[] content;
719 	}
720 	//matrix size
721 	uinteger matrixHeight, matrixWidth;
722 	//next write position
723 	uinteger xPosition, yPosition;
724 	
725 	LinkedList!Display toUpdate;
726 	
727 public:
728 	this(uinteger width, uinteger height){
729 		toUpdate = new LinkedList!Display;
730 		//set size
731 		matrixHeight = 1;
732 		matrixWidth = 1;
733 		if (width > 1){
734 			matrixWidth = width;
735 		}
736 		if (height > 1){
737 			matrixHeight = height;
738 		}
739 		//set write positions
740 		xPosition = 0;
741 		yPosition = 0;
742 	}
743 	~this(){
744 		toUpdate.destroy;
745 	}
746 	///Clear the matrix, resets write position
747 	void clear(){
748 		toUpdate.clear;
749 		resetWritePosition();
750 	}
751 	/// clears the matrix, resets write position, and changes matrix size
752 	void changeSize(uinteger width, uinteger height){
753 		clear;
754 		matrixWidth = width;
755 		matrixHeight = height;
756 		if (matrixWidth == 0){
757 			matrixWidth = 1;
758 		}
759 		if (matrixHeight == 0){
760 			matrixHeight = 1;
761 		}
762 	}
763 	///sets write position to (0, 0)
764 	void resetWritePosition(){
765 		xPosition = 0;
766 		yPosition = 0;
767 	}
768 	/// To write to terminal
769 	void write(char[] c, RGBColor textColor, RGBColor bgColor){
770 		//get available cells
771 		uinteger cells = (matrixHeight - yPosition) * matrixWidth;// first get the whole area
772 		cells -= xPosition;//subtract partial-lines
773 		if (c.length > cells){
774 			c = c[0 .. cells];
775 		}
776 		if (c.length > 0){
777 			uinteger toAdd = xPosition + (yPosition * matrixWidth);
778 			Display disp;
779 			disp.bgColor = bgColor;
780 			disp.textColor = textColor;
781 			for (uinteger i = 0, readFrom = 0, end = c.length+1; i < end; i ++){
782 				if ((i+toAdd)%matrixWidth == 0 || i == c.length){
783 					//line ended, append it
784 					disp.x = (readFrom+toAdd)%matrixWidth;
785 					disp.y = (readFrom+toAdd)/matrixWidth;
786 					disp.content = c[readFrom .. i].dup;
787 					if (disp.content != null){
788 						toUpdate.append(disp);
789 						readFrom = i;
790 					}
791 				}
792 			}
793 			// update x and y positions
794 			xPosition = (c.length+toAdd)%matrixWidth;
795 			yPosition = (c.length+toAdd)/matrixWidth;
796 		}
797 	}
798 	/// Changes the matrix's colors. Must be called before any writing has taken place
799 	void setColors(RGBColor textColor, RGBColor bgColor){
800 		toUpdate.clear;// because it's contents are going to be overwritten, so why bother waste time writing them?
801 		Display disp;
802 		disp.bgColor = bgColor;
803 		disp.textColor = textColor;
804 		disp.x = 0;
805 		disp.y = 0;
806 		disp.content.length = matrixWidth;
807 		disp.content[] = ' ';
808 		// loop and set colors
809 		for (uinteger i = 0; i < matrixHeight; i ++){
810 			toUpdate.append(disp);
811 			disp.y ++;
812 		}
813 	}
814 	///move to a different position to write
815 	void moveTo(uinteger x, uinteger y){
816 		if (x < matrixWidth && y < matrixHeight){
817 			xPosition = x;
818 			yPosition = y;
819 		}
820 	}
821 	///returns number of rows/lines in matrix
822 	@property uinteger rowCount(){
823 		return matrixHeight;
824 	}
825 	///returns number of columns in matrix
826 	@property uinteger colCount(){
827 		return matrixWidth;
828 	}
829 	///returns the point ox x-axis where next write will start from
830 	@property Position writePos(){
831 		Position r;
832 		r.x = xPosition;
833 		r.y = yPosition;
834 		return r;
835 	}
836 	/// Returns in an array, list of all changes that have to be drawn on terminal. Used by `Matrix.insert` to copy contents
837 	Display[] toArray(){
838 		return toUpdate.toArray;
839 	}
840 	///insert a matrix into this one at a position
841 	void insert(Matrix toInsert, uinteger x, uinteger y){
842 		Display[] newMatrix = toInsert.toArray.dup;
843 		// go through the new Displays and increae their x and y, and append them
844 		foreach (disp; newMatrix){
845 			disp.y += y;
846 			disp.x += x;
847 			
848 			toUpdate.append(disp);
849 		}
850 	}
851 	///Write contents of matrix to a QTerminal
852 	void flushToTerminal(QTerminal terminal){
853 		if (toUpdate.count > 0){
854 			toUpdate.resetRead;
855 			Display* disp = toUpdate.read;
856 			do{
857 				terminal.moveTo(cast(int)(*disp).x, cast(int)(*disp).y);
858 				terminal.setColors((*disp).textColor, (*disp).bgColor);
859 				terminal.writeChars((*disp).content);
860 				
861 				disp = toUpdate.read;
862 			}while (disp !is null);
863 			terminal.flush();
864 			
865 			toUpdate.clear;
866 		}
867 	}
868 }

871 ///Center-aligns text, returns that in an char[] with width as length. The empty part filled with ' '
872 char[] centerAlignText(char[] text, uinteger width, char fill = ' '){
873 	char[] r;
874 	if (text.length < width){
875 		r.length = width;
876 		uinteger offset = (width - text.length)/2;
877 		r[0 .. offset] = fill;
878 		r[offset .. offset+text.length][] = text;
879 		r[offset+text.length .. r.length] = fill;
880 	}else{
881 		r = text[0 .. width];
882 	}
883 	return r;
884 }

897 ///Converts hex color code to RGBColor
898 RGBColor hexToColor(string hex){
899 	RGBColor r;
900 	uinteger den = hexToDenary(hex);
901 	//min val for red in denary = 65536
902 	//min val for green in denary = 256
903 	//the remaining value is blue
904 	if (den >= 65536){
905 		r.r = cast(ubyte)((den / 65536));
906 		den -= r.r*65536;
907 	}
908 	if (den >= 256){
909 		r.g = cast(ubyte)((den / 256));
910 		den -= r.g*256;
911 	}
912 	r.b = cast(ubyte)den;
913 	return r;
914 }

916 unittest{
917 	RGBColor c;
918 	c.r = 10;
919 	c.g = 15;
920 	c.b = 0;
921 	assert("0A0F00".hexToColor == c);
922 }
