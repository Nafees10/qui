40 ///Key press event, keyboardEvent function is called with this

42 ///A note: backspace (`\b`) and enter (`\n`) are not included in KeyPress.NonCharKey

46 	/// Returns true if the key was a character.
47 	/// 
48 	/// A note: Enter/Return ('\n') is not included in KeyPress.NonCharKey

60 		F6 = 0x75 + 0xF0000,
61 		F7 = 0x76 + 0xF0000,
62 		F8 = 0x77 + 0xF0000,
63 		F9 = 0x78 + 0xF0000,
64 		F10 = 0x79 + 0xF0000,
65 		F11 = 0x7A + 0xF0000,
66 		F12 = 0x7B + 0xF0000,
67 		LeftArrow = 0x25 + 0xF0000,
68 		RightArrow = 0x27 + 0xF0000,
69 		UpArrow = 0x26 + 0xF0000,
70 		DownArrow = 0x28 + 0xF0000,

116 			return false;
117 		}

437 ///Used to place widgets in an order (i.e vertical or horizontal)
438 ///
439 ///The QTerminal is also a layout, basically.
440 ///
441 ///Name in theme: 'layout';
442 class QLayout : QWidget{
443 private:
444 	// array of all the widgets that have been added to this layout
445 	QWidget[] widgetList;
446 	// stores the layout type, horizontal or vertical
447 	LayoutDisplayType layoutType;
448 	// stores whether an update is in progress
449 	bool isUpdating = false;
450 	
451 	// recalculates the size of every widget inside layout
452 	void recalculateWidgetsSize(LayoutDisplayType T)(QWidget[] widgets, uinteger totalSpace, uinteger totalRatio){
453 		static if (T != LayoutDisplayType.Horizontal && T != LayoutDisplayType.Vertical){
454 			assert(false);
455 		}
456 		bool repeat;
457 		do{
458 			repeat = false;
459 			foreach(i, widget; widgets){
460 				if (widget.visible){
461 					// calculate width or height
462 					uinteger newSpace = ratioToRaw(widget.sizeRatio, totalRatio, totalSpace);
463 					uinteger calculatedSpace = newSpace;
464 					//apply size
465 					static if (T == LayoutDisplayType.Horizontal){
466 						widget.size.height = widgetSize.height;
467 						widget.size.width = newSpace;
468 						newSpace = widget.size.width;
469 					}else{
470 						widget.size.width = widgetSize.width;
471 						widget.size.height = newSpace;
472 						newSpace = widget.size.height;
473 					}
474 					if (newSpace != calculatedSpace){
475 						totalRatio -= widget.sizeRatio;
476 						totalSpace -= newSpace;
477 						widgets = widgets.dup.deleteElement(i);
478 						repeat = true;
479 						break;
480 					}
481 					// check if there's enough space to contain that widget
482 					if (newSpace > totalSpace){
483 						newSpace = 0;
484 						widget.visible = false;
485 					}
486 				}
487 			}
488 		} while (repeat);
489 	}
490 	/// calculates and assigns widgets positions based on their sizes
491 	void recalculateWidgetsPosition(LayoutDisplayType T)(QWidget[] widgets){
492 		static if (T != LayoutDisplayType.Horizontal && T != LayoutDisplayType.Vertical){
493 			assert(false);
494 		}
495 		uinteger previousSpace = (T == LayoutDisplayType.Horizontal ? widgetPosition.x : widgetPosition.y);
496 		uinteger fixedPoint = (T == LayoutDisplayType.Horizontal ? widgetPosition.y : widgetPosition.x);
497 		foreach(widget; widgets){
498 			if (widget.visible){
499 				static if (T == LayoutDisplayType.Horizontal){
500 					widget.position.y = widgetPosition.y;
501 					widget.position.x = previousSpace;
502 					previousSpace += widget.size.width;
503 				}else{
504 					widget.position.x = widgetPosition.x;
505 					widget.position.y = previousSpace;
506 					previousSpace += widget.size.height;
507 				}
508 			}
509 		}
510 	}
511 	
512 public:
513 	this(LayoutDisplayType type){
514 		layoutType = type;
515 		widgetWantsInput = false;
516 	}
517 	
518 	/// adds (appends) a widget to the widgetList, and makes space for it
519 	/// 
520 	/// If there a widget is too large, it's marked as not visible
521 	void addWidget(QWidget widget){
522 		widget.setTermInterface = termInterface;
523 		//add it to array
524 		widgetList ~= widget;
525 		//recalculate all widget's size to adjust
526 		resizeEvent();
527 	}
528 	/// adds (appends) widgets to the widgetList, and makes space for them
529 	/// 
530 	/// If there a widget is too large, it's marked as not visible
531 	void addWidget(QWidget[] widgets){
532 		foreach(widget; widgets){
533 			widget.setTermInterface = termInterface;
534 		}
535 		// add to array
536 		widgetList ~= widgets.dup;
537 		//resize
538 		resizeEvent();
539 	}
540 	
541 	/// Recalculates size and position for all visible widgets
542 	/// If a widget is too large to fit in, it's visibility is marked false
543 	override void resizeEvent(){
544 		//disable update during size change, saves CPU and time
545 		super.resizeEvent();
546 		isUpdating = true;
547 		uinteger ratioTotal;
548 		foreach(w; widgetList){
549 			if (w.visible()){
550 				ratioTotal += w.sizeRatio;
551 			}
552 		}
553 		if (layoutType == LayoutDisplayType.Horizontal){
554 			recalculateWidgetsSize!(LayoutDisplayType.Horizontal)(widgetList, widgetSize.width, ratioTotal);
555 			recalculateWidgetsPosition!(LayoutDisplayType.Horizontal)(widgetList);
556 		}else{
557 			recalculateWidgetsSize!(LayoutDisplayType.Vertical)(widgetList, widgetSize.height, ratioTotal);
558 			recalculateWidgetsPosition!(LayoutDisplayType.Vertical)(widgetList);
559 		}
560 		foreach (widget; widgetList){
561 			widget.resizeEvent;
562 		}
563 		isUpdating = false;
564 	}
565 	override bool update(Matrix display){
566 		bool updated = false;
567 		//check if already updating, case yes, return false
568 		if (!isUpdating){
569 			isUpdating = true;
570 			//go through all widgets, check if they need update, update them
571 			Matrix wDisplay = new Matrix(1,1);
572 			foreach(widget; widgetList){
573 				if (widget.visible){
574 					wDisplay.changeSize(widget.size.width, widget.size.height);
575 					wDisplay.resetWritePosition();
576 					if (widget.update(wDisplay)){
577 						display.insert(wDisplay, widget.position.x - this.position.x,
578 							widget.position.y - this.position.y);
579 						updated = true;
580 					}
581 					wDisplay.clear();
582 				}
583 			}
584 			isUpdating = false;
585 		}else{
586 			return false;
587 		}
588 		return updated;
589 	}
590 
591 	// override setTermInterface to change it for all child widgets as well
592 	/// 
593 	/// **Should __NEVER__ be called from outside, only the owner should call this**
594 	override public @property QTermInterface setTermInterface(QTermInterface newInterface){
595 		// change it for itself
596 		auto r = super.setTermInterface(newInterface);
597 		// change it for all child widgets
598 		foreach(widget; widgetList){
599 			widget.setTermInterface = termInterface;
600 		}
601 		// register itself
602 		termInterface.registerWidget(this);
603 		return r;
604 	}
605 }

607 /// Provides interface to the QTerminal for widgets
608 struct QTermInterface{
609 	/// the terminal to provide interface to
610 	private QTerminal term;
611 	/// constructor
612 	this (QTerminal terminal){
613 		term = terminal;
614 	}
615 	/// called to set position of the cursor
616 	/// 
617 	/// Returns: true on success, false on failure, probably because caller isnt activeWidget
618 	bool setCursorPos(Position cursorPos, QWidget callerWidget){
619 		if (term)
620 			return term.setCursorPos(cursorPos.x, cursorPos.y, callerWidget);
621 		return false;
622 	}
623 	/// used to set "hotkey", so each a key is pressed, a specific widget's keyboardEvent will be triggered
624 	/// 
625 	/// Returns: true on success, false on failure, probably because the key is already registered
626 	bool setKeyHandler(KeyPress key, QWidget handlerWidget){
627 		if (term)
628 			return term.registerKeyHandler(key, handlerWidget);
629 		return false;
630 	}
631 	/// registers a new widget with the terminal. For a widget to be active, it must be registered first
632 	void registerWidget(QWidget newWidget){
633 		if (term)
634 			term.registerWidget(newWidget);
635 	}
636 	/// forces an update of the terminal, only widgets that need update will be updated.
637 	/// 
638 	/// Returns: true if at least one widget was updated, otherwise, false
639 	bool forceUpdate(){
640 		if (term)
641 			return term.updateDisplay;
642 		return false;
643 	}
644 	/// Returns: true if a widget is active widget
645 	bool isActiveWidget(QWidget widget){
646 		if (term){
647 			return term.isActiveWidget(widget);
648 		}
649 		return false;
650 	}
651 }

813 	override public void mouseEvent(MouseClick mouse){
814 		super.mouseEvent(mouse);
815 		QWidget lastActiveWidget = activeWidget;
816 		activeWidget = null;
817 		activeWidgetIndex = -1;
818 		foreach (i, widget; registeredWidgets){
819 			if (widget.visible && widget.wantsInput){
820 				Position p = widget.position;
821 				Size s = widget.size;
822 				//check x-axis
823 				if (mouse.x >= p.x && mouse.x < p.x + s.width){
824 					//check y-axis
825 					if (mouse.y >= p.y && mouse.y < p.y + s.height){
826 						//mark this widget as active
827 						activeWidget = widget;
828 						activeWidgetIndex = i;
829 						// make mouse position relative to widget position, not 0:0
830 						mouse.x = mouse.x - activeWidget.position.x;
831 						mouse.y = mouse.y - activeWidget.position.y;
832 						//call mouseEvent
833 						widget.mouseEvent(mouse);
834 						break;
835 					}
836 				}
837 			}
838 		}
839 		if (activeWidget != lastActiveWidget){
840 			if (lastActiveWidget){
841 				lastActiveWidget.activateEvent(false);
842 			}
843 			if (activeWidget){
844 				activeWidget.activateEvent(true);
845 			}
846 		}
847 	}

849 	override public void keyboardEvent(KeyPress key){
850 		super.keyboardEvent(key);
851 		// check if the activeWidget wants Tab, otherwise, if is Tab, make the next widget active
852 		if (key.key == KeyPress.NonCharKey.Escape || (key.key == '\t' && (activeWidgetIndex < 0 || !activeWidget.wantsTab))){
853 			QWidget lastActiveWidget = activeWidget;
854 			// make the next widget active
855 			if (registeredWidgets.length > 0){
856 				activeWidgetIndex ++;
857 				if (activeWidgetIndex > registeredWidgets.length){
858 					activeWidgetIndex = 0;
859 				}
860 				// see if it wants input, case no, switch to some other widget
861 				for (;activeWidgetIndex < registeredWidgets.length; activeWidgetIndex ++){
862 					if (registeredWidgets[activeWidgetIndex].wantsInput && registeredWidgets[activeWidgetIndex].visible){
863 						break;
864 					}
865 				}
866 				if (activeWidgetIndex < registeredWidgets.length){
867 					activeWidget = registeredWidgets[activeWidgetIndex];
868 				}else{
869 					activeWidget = null;
870 					activeWidgetIndex = -1;
871 				}
872 			}else{
873 				activeWidgetIndex = -1;
874 				activeWidget = null;
875 			}
876 			if (activeWidget != lastActiveWidget){
877 				if (lastActiveWidget){
878 					lastActiveWidget.activateEvent(false);
879 				}
880 				if (activeWidget){
881 					activeWidget.activateEvent(true);
882 				}
883 			}
884 		}else if (key in keysToCatch){
885 			// this is a registered key, only a specific widget catches it
886 			keysToCatch[key].keyboardEvent(key);
887 		}else if (activeWidget !is null){
888 			activeWidget.keyboardEvent (key);
889 		}
890 	}

893 	/// starts the UI loop
894 	void run(){
895 		InputEvent event;
896 		isRunning = true;
897 		//resize all widgets
898 		resizeEvent();
899 		//draw the whole thing
900 		updateDisplay();
901 		while (isRunning){
902 			event = input.nextEvent;
903 			//check event type
904 			if (event.type == event.Type.KeyboardEvent){
905 				// prevent any updates during event, the update will be done at end
906 				isUpdating = true;
907 				KeyPress kPress;
908 				kPress.key = event.get!(event.Type.KeyboardEvent).which;
909 				this.keyboardEvent(kPress);
910 				// now it can update
911 				isUpdating = false;
912 				updateDisplay;
913 			}else if (event.type == event.Type.MouseEvent){
914 				// prevent any updates during event, the update will be done at end
915 				isUpdating = true;
916 				MouseEvent mEvent = event.get!(event.Type.MouseEvent);
917 				MouseClick mPos;
918 				mPos.x = mEvent.x;
919 				mPos.y = mEvent.y;
920 				switch (mEvent.buttons){
921 					case MouseEvent.Button.Left:
922 						mPos.mouseButton = mPos.Button.Left;
923 						break;
924 					case MouseEvent.Button.Right:
925 						mPos.mouseButton = mPos.Button.Right;
926 						break;
927 					case MouseEvent.Button.ScrollUp:
928 						mPos.mouseButton = mPos.Button.ScrollUp;
929 						break;
930 					case MouseEvent.Button.ScrollDown:
931 						mPos.mouseButton = mPos.Button.ScrollDown;
932 						break;
933 					default:
934 						continue;
935 				}
936 				this.mouseEvent(mPos);
937 				// now it can update
938 				isUpdating = false;
939 				updateDisplay;
940 			}else if (event.type == event.Type.SizeChangedEvent){
941 				//change matrix size
942 				termDisplay.changeSize(cast(uinteger)terminal.width, cast(uinteger)terminal.height);
943 				termDisplay.setColors(textColor, backgroundColor);
944 				//update self size
945 				terminal.updateSize;
946 				widgetSize.height = terminal.height;
947 				widgetSize.width = terminal.width;
948 				this.clear;
949 				//call size change on all widgets
950 				resizeEvent();
951 				updateDisplay;
952 			}else if (event.type == event.Type.UserInterruptionEvent || event.type == event.Type.HangupEvent){
953 				//die here
954 				terminal.clear;
955 				isRunning = false;
956 				break;
957 			}
958 		}
959 		//in case an exception prevents it from being set to false before
960 		isRunning = false;
961 	}

965 		isRunning = false;

975 class Matrix{
976 private:
977 	struct Display{
978 		uinteger x, y;
979 		RGB bgColor, textColor;
980 		char[] content;

982 	//matrix size
983 	uinteger matrixHeight, matrixWidth;
984 	//next write position

987 	LinkedList!Display toUpdate;
988 	
989 public:
990 	this(uinteger width, uinteger height){
991 		toUpdate = new LinkedList!Display;
992 		//set size
993 		matrixHeight = 1;
994 		matrixWidth = 1;
995 		if (width > 1){
996 			matrixWidth = width;
997 		}
998 		if (height > 1){
999 			matrixHeight = height;
1000 		}
1001 		//set write positions
1002 		xPosition = 0;
1003 		yPosition = 0;
1004 	}
1005 	~this(){
1006 		toUpdate.destroy;
1007 	}
1008 	///Clear the matrix, resets write position
1009 	void clear(){
1010 		toUpdate.clear;
1011 		resetWritePosition();
1012 	}
1013 	/// clears the matrix, resets write position, and changes matrix size
1014 	void changeSize(uinteger width, uinteger height){
1015 		clear;
1016 		matrixWidth = width;
1017 		matrixHeight = height;
1018 		if (matrixWidth == 0){
1019 			matrixWidth = 1;
1020 		}
1021 		if (matrixHeight == 0){
1022 			matrixHeight = 1;
1023 		}
1024 	}
1025 	///sets write position to (0, 0)
1026 	void resetWritePosition(){
1027 		xPosition = 0;
1028 		yPosition = 0;
1029 	}
1030 	/// To write to terminal
1031 	void write(char[] c, RGB textColor, RGB bgColor){
1032 		//get available cells
1033 		uinteger cells = (matrixHeight - yPosition) * matrixWidth;// first get the whole area
1034 		cells -= xPosition;//subtract partial-lines
1035 		if (c.length > cells){
1036 			c = c[0 .. cells];
1037 		}
1038 		if (c.length > 0){
1039 			uinteger toAdd = xPosition + (yPosition * matrixWidth);
1040 			Display disp;
1041 			disp.bgColor = bgColor;
1042 			disp.textColor = textColor;
1043 			for (uinteger i = 0, readFrom = 0, end = c.length+1; i < end; i ++){
1044 				if ((i+toAdd)%matrixWidth == 0 || i == c.length){
1045 					//line ended, append it
1046 					disp.x = (readFrom+toAdd)%matrixWidth;
1047 					disp.y = (readFrom+toAdd)/matrixWidth;
1048 					disp.content = c[readFrom .. i].dup;
1049 					if (disp.content != null){
1050 						toUpdate.append(disp);
1051 						readFrom = i;
1052 					}
1053 				}
1054 			}
1055 			// update x and y positions
1056 			xPosition = (c.length+toAdd)%matrixWidth;
1057 			yPosition = (c.length+toAdd)/matrixWidth;
1058 		}
1059 	}

1063 		char[] toFillWith;
1064 		toFillWith.length = matrixWidth - xPosition;
1065 		toFillWith[] = c;

1121 	///insert a matrix into this one at a position
1122 	void insert(Matrix toInsert, uinteger x, uinteger y){
1123 		Display[] newMatrix = toInsert.toArray.dup;
1124 		// go through the new Displays and increae their x and y, and append them
1125 		foreach (disp; newMatrix){

1133 	void flushToTerminal(QTerminal terminal){

1135 			toUpdate.resetRead;
1136 			Display* disp = toUpdate.read;

1144 			terminal.flush();
1145 			
1146 			toUpdate.clear;
1147 		}
1148 	}

1178 ///Converts hex color code to RGB
1179 RGB hexToColor(string hex){
1180 	RGB r;
1181 	uinteger den = hexToDenary(hex);
1182 	//min val for red in denary = 65536
1183 	//min val for green in denary = 256
1184 	//the remaining value is blue
1185 	if (den >= 65536){
1186 		r.r = cast(ubyte)((den / 65536));
1187 		den -= r.r*65536;
1188 	}
1189 	if (den >= 256){
1190 		r.g = cast(ubyte)((den / 256));
1191 		den -= r.g*256;
1192 	}
1193 	r.b = cast(ubyte)den;
1194 	return r;
1195 }
